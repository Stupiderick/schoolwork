### ch5-p2
if the input size is larger than the buffer, gets() may corrupt the other variables.


### ch5-p3
#include <stdio.h>

int main()
{
    char ch1[10], ch2[10];
    int num;

    char * ch = "hello 3 world";
    sscanf(ch, "%s %d %s", ch1, &num, ch2);

    return 0;
}

### ch5-p5
#define _GNU_SOURCE
#include <stdio.h>
#include <stdlib.h>

int main() {
    char file[] = "file.txt";
    FILE *myfile = fopen(file, "r");
    char *buffer = NULL;
    size_t capacity = 0;

    while (getline(&buffer, &capacity, myfile) != -1) {
        printf("%s", buffer);
    }

    free(buffer);
    fclose(myfile);
    return 0;
}
